1. 輸入三角形的底、高, 求面積
#include <bits/stdc++.h>
using namespace std;
using lint = long long int;
#define endl "\n";

int main()
{
    cout.sync_with_stdio(false);
    cin.tie(nullptr);

    float w, h;
    cin >> w >> h;
    cout << (w * h / 2.0);
}
